import random

def start_game():

    width = int(input("Enter the width of the game board: "))
    height = int(input("Enter the height of the game board: "))

def get_food_coordinates(width, height):
    while True:
        x = random.randint(0, width - 1)
        y = random.randint(0, height - 1)
        food_coordinates = (x, y)
        if food_coordinates not in snake:
            return food_coordinates

def draw_board(width, height, snake, food):
    for y in range(height):
        row = ""
        for x in range(width):
            if (x, y) == food:
                row += "F"
            elif (x, y) in snake:
                row += "S"
            else:
                row += " "
        print(row)

def handle_input(direction, snake):
    if direction == "up":
        snake.append((snake[-1][0], snake[-1][1] - 1))
    elif direction == "down":
        snake.append((snake[-1][0], snake[-1][1] + 1))
    elif direction == "left":
        snake.append((snake[-1][0] - 1, snake[-1][1]))
    elif direction == "right":
        snake.append((snake[-1][0] + 1, snake[-1][1]))

def main():
    width, height, snake, food, direction = start_game()
    draw_board(width, height, snake, food)

    while True:
        draw_board(width, height, snake, food)
        handle_input(direction, snake)
        if snake[-1] in snake[:-1]:
            print("Game Over!")
            break

if __name__ == "__main__":
    main()
